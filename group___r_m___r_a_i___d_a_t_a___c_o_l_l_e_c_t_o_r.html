Detailed Description

Middleware to implement the Data Collector for Reality AI applications. This module implements the Data Collector Interface.

Overview

Data Collector is to abstract the collection of data from sensors so that samples are collected and accumulated into fixed length frames before being made available to upper modules/application. Support of "snapshot" mode and "data feed" mode are required to accommodate for background and cooperative data collection. Each sensor will be captured into a separate frame buffer. Frame buffers are allocated by users, and they shall have the same amount of data samples (int32_t, float, uint8_t etc). User application has to make sure that each frame buffer will be filled up at the same rate. PING-PONG buffer is required for seamless operation.

When all frame buffers are filled up, they will be provided to the upper modules/application via data ready callback. When they are consumed, upper module has to release them. Ideally buffers will be released before the other set of buffers are filled up. However, it is possible that frame buffers will overrun due to the fact that upper module may take longer time to process the data in some cases. If it happens, application will be notified via the error callback. No intervention is required from user side in this case. Buffer overrun will go away when frame buffers are released. However, if there is a buffer-out-of-sync error, users need to find out whether all sensors work at the same pace.

Features

  • Snapshot mode and data feed mode are supported
  • Maximumlly 8 sensors are supported for each mode
  • Mix mode is supported(Snapshot mode and data feed mode work symultaneously)

Snapshot Mode

Snapshot mode will periodically pull data from the user-specified places and save to designated frame buffers in the background. Renesas MCUs are well equipped to support this mode. DTC with its chain mode enables data collection from various, potentially non-linear and different-sized sources, while the General PWM Timer (GPT) and the Asynchronous General Purpose Timer provides periodic interrupt that can be used as the activation source of DTC. When sensor source addresses are reigstered, application needs to start the timer so that DTC will start data collection.

Data Feed Mode

Data feed mode will require data producer to push data directly to the designated frame buffer whenever data is ready. Data can be pushed synchronously or asynchronously. Synchronous mode is for the use case that the data producer has a short amount of data to be copied to the frame buffer. Asynchronous mode is for dmac/dtc transfer. Application is responsible to initialize dmac/dtc transfer descriptors.

Note when work in mixed mode, users must take care to make sure snapshot mode channels and data mode channels work at the same pace.

Examples

Basic Example

This is a basic example of minimal use of the Data Collector implementation in an application.

void rm_rai_data_collector_example();
void data_ready_callback_test(const rai_data_collector_callback_args_t * p_args);
void error_callback_test(const rai_data_collector_error_callback_args_t * p_args);
void dma_callback(const void * p_args);
void timer_callback(const timer_callback_args_t * p_args);
#define DC_TEST_REQUIRED_FRAME_LEN (256U)
#define DC_TEST_PING_PONG_BUFFER_LEN (DC_TEST_REQUIRED_FRAME_LEN * 2U)
#define DC_TEST_CHAN_BUFFER_LEN (64U)
#define DC_TEST_SNAPSHOT_CHAN_COUNT (3)
#define DC_TEST_DATA_FEED_CHAN_COUNT (3)
#define DC_TEST_CHAN_COUNT (DC_TEST_SNAPSHOT_CHAN_COUNT + DC_TEST_DATA_FEED_CHAN_COUNT)
#define DC_TEST_VIRTUAL_CHAN_COUNT (DC_TEST_DATA_FEED_CHAN_COUNT + (DC_TEST_SNAPSHOT_CHAN_COUNT > 0 ? 1 : 0))
extern const timer_instance_t g_timer0;
extern const transfer_instance_t g_transfer0;
static uint8_t g_frame_buf[DC_TEST_CHAN_COUNT][DC_TEST_PING_PONG_BUFFER_LEN];
static uint8_t g_chan_buf[DC_TEST_CHAN_COUNT][DC_TEST_CHAN_BUFFER_LEN];
static uint8_t g_data_feed_sync_buf[DC_TEST_CHAN_BUFFER_LEN];
static uint8_t * g_data_feed_async_dma_buf;
static rai_data_collector_error_event_t g_dc_error_event = RAI_DATA_COLLECTOR_ERROR_TYPE_NONE;
static volatile bool g_exit = false;
const rai_data_collector_snapshot_cfg_t g_ss_cfg =
{
.channels = DC_TEST_SNAPSHOT_CHAN_COUNT,
.transfer_len = DC_TEST_CHAN_BUFFER_LEN,
.p_timer = &g_timer0,
.p_transfer = &g_transfer0,
};
const rai_data_collector_data_feed_cfg_t g_df_cfg =
{
.channels = DC_TEST_DATA_FEED_CHAN_COUNT,
};
/* Initialize g_extended_cfg */
static transfer_info_t transfer_info[DC_TEST_SNAPSHOT_CHAN_COUNT];
static rai_data_collector_frame_buffer_t ping_pong_buf[DC_TEST_CHAN_COUNT];
static rai_data_collector_frame_buffer_handle_t ping_pong_buf_hnd[DC_TEST_VIRTUAL_CHAN_COUNT];
const rai_data_collector_extended_cfg_t g_extended_cfg =
{
.p_ping_pong_buf = ping_pong_buf, // Frame buffer used as PING-PONG buffer
.p_ping_pong_buf_hnd = ping_pong_buf_hnd,
.p_transfer_info = transfer_info,
};
rai_data_collector_cfg_t g_dc_cfg =
{
.p_snapshot_cfg = &g_ss_cfg,
.p_data_feed_cfg = &g_df_cfg,
.channels = DC_TEST_CHAN_COUNT,
.virt_channels = DC_TEST_VIRTUAL_CHAN_COUNT,
.channel_ready_mask = 0,
.required_frame_len = DC_TEST_REQUIRED_FRAME_LEN,
.p_callback = data_ready_callback_test,
.p_error_callback = error_callback_test,
.p_context = NULL
};
rai_data_collector_instance_t g_dc_ctrl;
void rm_rai_data_collector_example ()
{
fsp_err_t err;
/* Initialize channel frame buffers */
for (int32_t i = 0; i < DC_TEST_CHAN_COUNT; i++)
{
ping_pong_buf[i].p_buf = (void *) g_frame_buf[i];
ping_pong_buf[i].data_type = RAI_DATA_COLLECTOR_DATA_TYPE_UINT8_T;
}
/* Initialize channel ready mask */
for (int32_t i = 0; i < DC_TEST_VIRTUAL_CHAN_COUNT; i++)
{
g_dc_cfg.channel_ready_mask |= 1 << i;
}
/* Initialize transfer info */
for (int32_t i = 0; i < DC_TEST_SNAPSHOT_CHAN_COUNT; i++)
{
transfer_info[i].transfer_settings_word_b.dest_addr_mode = TRANSFER_ADDR_MODE_INCREMENTED;
transfer_info[i].transfer_settings_word_b.src_addr_mode = TRANSFER_ADDR_MODE_INCREMENTED;
transfer_info[i].transfer_settings_word_b.repeat_area = TRANSFER_REPEAT_AREA_SOURCE;
transfer_info[i].transfer_settings_word_b.irq = TRANSFER_IRQ_END;
transfer_info[i].transfer_settings_word_b.size = TRANSFER_SIZE_1_BYTE; // data type is uint8_t
transfer_info[i].transfer_settings_word_b.chain_mode = TRANSFER_CHAIN_MODE_EACH;
transfer_info[i].transfer_settings_word_b.mode = TRANSFER_MODE_BLOCK;
transfer_info[i].p_dest = ping_pong_buf[i + DC_TEST_DATA_FEED_CHAN_COUNT].p_buf;
transfer_info[i].p_src = (void const *) NULL;
transfer_info[i].num_blocks = 1;
transfer_info[i].length = g_ss_cfg.transfer_len;
}
transfer_info[DC_TEST_SNAPSHOT_CHAN_COUNT - 1].transfer_settings_word_b.chain_mode = TRANSFER_CHAIN_MODE_DISABLED;
err = RM_RAI_DATA_COLLECTOR_Open(&g_dc_ctrl, &g_dc_cfg);
/* Handle any errors. This function should be defined by the user. */
assert(FSP_SUCCESS == err);
for (uint8_t i = 0; i < DC_TEST_SNAPSHOT_CHAN_COUNT; i++)
{
RM_RAI_DATA_COLLECTOR_SnapshotChannelRegister(&g_dc_ctrl, i, g_chan_buf[i + DC_TEST_DATA_FEED_CHAN_COUNT]);
/* Handle any errors. This function should be defined by the user. */
assert(FSP_SUCCESS == err);
}
err = RM_RAI_DATA_COLLECTOR_SnapshotStart(&g_dc_ctrl);
/* Handle any errors. This function should be defined by the user. */
assert(FSP_SUCCESS == err);
RM_RAI_DATA_COLLECTOR_ChannelBufferGet(&g_dc_ctrl, 0, (void **) &g_data_feed_async_dma_buf);
/* use g_data_feed_async_dma_buf for dma transfer */
do
{
/* Process */
/* Process error event */
if (RAI_DATA_COLLECTOR_ERROR_TYPE_BUF_OUT_OF_SYNC == g_dc_error_event)
{
RM_RAI_DATA_COLLECTOR_SnapshotStop(&g_dc_ctrl);
/* Stop data feed mode channels */
RM_RAI_DATA_COLLECTOR_BufferReset(&g_dc_ctrl);
RM_RAI_DATA_COLLECTOR_SnapshotStart(&g_dc_ctrl);
/* Start data feed mode channels */
RM_RAI_DATA_COLLECTOR_ChannelBufferGet(&g_dc_ctrl, 0, (void **) &g_data_feed_async_dma_buf);
/* use g_data_feed_async_dma_buf for dma transfer */
}
} while (!g_exit);
err = RM_RAI_DATA_COLLECTOR_SnapshotStop(&g_dc_ctrl);
/* Handle any errors. This function should be defined by the user. */
assert(FSP_SUCCESS == err);
err = RM_RAI_DATA_COLLECTOR_Close(&g_dc_ctrl);
/* Handle any errors. This function should be defined by the user. */
assert(FSP_SUCCESS == err);
}
void dma_callback (const void * p_args)
{
FSP_PARAMETER_NOT_USED(p_args);
/* Prepare for next transfer.*/
RM_RAI_DATA_COLLECTOR_ChannelBufferGet(&g_dc_ctrl, 0, (void **) &g_data_feed_async_dma_buf);
/* use g_data_feed_async_dma_buf for dma transfer */
}
void timer_callback (const timer_callback_args_t * p_args)
{
FSP_PARAMETER_NOT_USED(p_args);
RM_RAI_DATA_COLLECTOR_ChannelWrite(&g_dc_ctrl, 0, (void *) g_data_feed_sync_buf, DC_TEST_CHAN_BUFFER_LEN);
}
/* Data ready callback called when all sensor data are ready. */
void data_ready_callback_test (const rai_data_collector_callback_args_t * p_args)
{
FSP_PARAMETER_NOT_USED(p_args);
/* TO DO: consume frame buffers.
* Note this callback may be called in ISR.Please return as fast as possible.
*/
}
/* Error callback when there is buffer overflow. */
void error_callback_test (const rai_data_collector_error_callback_args_t * p_args)
{
/* TO DO */
FSP_PARAMETER_NOT_USED(p_args);
g_dc_error_event = p_args->event;
}

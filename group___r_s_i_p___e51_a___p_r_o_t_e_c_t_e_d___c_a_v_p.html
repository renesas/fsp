Overview

This module provides RSIP-E51A CAVP Certified functions.

HW Overview

RSIP-E51A Protected Mode CAVP certification has been obtained using the RA8M1 as the target processor.

Crypto Peripheral version Devices
RSIP-E51A RA8M1, RA8D1, RA8T1

State Transition

This driver has the following states.

State Name Details
STATE_INITIAL Driver is not open.
STATE_MAIN Ready to Compute.
STATE_AES Computing AES (unauthenticated cipher)
STATE_AES_AEAD Computing AES (AEAD).
STATE_AES_MAC Computing AES (MAC).
STATE_SHA Computing SHA.
STATE_HMAC Computing HMAC.
STATE_KDF_SHA Computing KDF SHA.
STATE_KDF_HMAC Computing KDF HMAC.

There are two types of APIs provided by the RSIP driver for accelerating cryptographic operations: those that provide cryptographic operations in a single API and those that provide them in multiple APIs. In this document, the former is referred to as single-part operations and the latter as multi-part operations. Each corresponds to the following algorithms:

  • Single-part operations: DRBG, Key management, ECC, RSA, SHA, HMAC
  • Multi-part operations: AES, SHA, HMAC

Multi-part operations are APIs which split a single cryptographic operation into a sequence of separate steps (e.g. Init-Update-Finish). This enables fine control over the configuration of the cryptographic operation, and allows the message data to be processed in fragments instead of all at once.

Due to the above characteristics, it is necessary to manage the operational states in the multi-part operations.

Examples

AES cipher Example

This is an example of AES-CBC encryption/decryption.

void r_rsip_example_aes ()
{
fsp_err_t err;
static const uint8_t plain[RSIP_PRV_BYTE_SIZE_AES_BLOCK * 2] BSP_ALIGN_VARIABLE(4) =
{
0x52, 0x65, 0x6e, 0x65, 0x73, 0x61, 0x73, 0x20, 0x45, 0x6c, 0x65, 0x63, 0x74, 0x72, 0x6f, 0x6e,
0x69, 0x63, 0x73, 0x20, 0x43, 0x6f, 0x72, 0x70, 0x6f, 0x72, 0x61, 0x74, 0x69, 0x6f, 0x6e, 0x00
};
uint8_t iv[RSIP_PRV_BYTE_SIZE_AES_BLOCK] BSP_ALIGN_VARIABLE(4) = {0};
uint8_t cipher_calculated[RSIP_PRV_BYTE_SIZE_AES_BLOCK * 2] BSP_ALIGN_VARIABLE(4) = {0};
uint8_t plain_calculated[RSIP_PRV_BYTE_SIZE_AES_BLOCK * 2] BSP_ALIGN_VARIABLE(4) = {0};
/* Prepare wrapped key data area */
uint8_t wrapped_key_value[RSIP_BYTE_SIZE_WRAPPED_KEY(RSIP_KEY_TYPE_AES_256)] BSP_ALIGN_VARIABLE(4);
rsip_wrapped_key_t wrapped_key =
{
.type = RSIP_KEY_TYPE_AES_256, .p_value = &wrapped_key_value
};
/* Initialize the driver */
err = R_RSIP_Open(&g_rsip_ctrl, &g_rsip_cfg);
assert(FSP_SUCCESS == err);
/* Generate a aes 256-bit key */
err = R_RSIP_KeyGenerate(&g_rsip_ctrl, &wrapped_key);
assert(FSP_SUCCESS == err);
/* Generate a nonce */
err = R_RSIP_RandomNumberGenerate(&g_rsip_ctrl, iv);
assert(FSP_SUCCESS == err);
/* Encrypt a plaintext */
err = R_RSIP_AES_Cipher_Init(&g_rsip_ctrl, RSIP_AES_CIPHER_MODE_CBC_ENC, &wrapped_key, iv);
assert(FSP_SUCCESS == err);
err = R_RSIP_AES_Cipher_Update(&g_rsip_ctrl, plain, cipher_calculated, RSIP_PRV_BYTE_SIZE_AES_BLOCK * 2);
assert(FSP_SUCCESS == err);
err = R_RSIP_AES_Cipher_Finish(&g_rsip_ctrl);
assert(FSP_SUCCESS == err);
/* Decrypt a ciphertext using the same key */
err = R_RSIP_AES_Cipher_Init(&g_rsip_ctrl, RSIP_AES_CIPHER_MODE_CBC_DEC, &wrapped_key, iv);
assert(FSP_SUCCESS == err);
err = R_RSIP_AES_Cipher_Update(&g_rsip_ctrl, cipher_calculated, plain_calculated, RSIP_PRV_BYTE_SIZE_AES_BLOCK);
assert(FSP_SUCCESS == err);
err = R_RSIP_AES_Cipher_Update(&g_rsip_ctrl,
cipher_calculated + RSIP_PRV_BYTE_SIZE_AES_BLOCK,
plain_calculated + RSIP_PRV_BYTE_SIZE_AES_BLOCK,
RSIP_PRV_BYTE_SIZE_AES_BLOCK);
assert(FSP_SUCCESS == err);
err = R_RSIP_AES_Cipher_Finish(&g_rsip_ctrl);
assert(FSP_SUCCESS == err);
/* Compare plain and plain_calculated */
assert(0 == memcmp(plain, plain_calculated, RSIP_PRV_BYTE_SIZE_AES_BLOCK * 2));
/* Close the driver */
err = R_RSIP_Close(&g_rsip_ctrl);
assert(FSP_SUCCESS == err);
}

AES-GCM Example

This is an example of AES-GCM encryption/decryption.

#define GCM_BLOCK_LEN (16)
#define GCM_TEXT_LEN (31)
#define GCM_NONCE_LEN (12)
#define GCM_AAD_LEN (8)
#define GCM_TAG_LEN (8)
void r_rsip_example_gcm ()
{
fsp_err_t err;
static const uint8_t plain[GCM_TEXT_LEN] BSP_ALIGN_VARIABLE(4) =
{
0x52, 0x65, 0x6e, 0x65, 0x73, 0x61, 0x73, 0x20, 0x45, 0x6c, 0x65, 0x63, 0x74, 0x72, 0x6f, 0x6e,
0x69, 0x63, 0x73, 0x20, 0x43, 0x6f, 0x72, 0x70, 0x6f, 0x72, 0x61, 0x74, 0x69, 0x6f, 0x6e,
};
static const uint8_t aad[GCM_AAD_LEN] BSP_ALIGN_VARIABLE(4) =
{
0x00, 0x01, 0x02, 0x03, 0x04, 0x05, 0x06, 0x07
};
uint8_t nonce[GCM_BLOCK_LEN] BSP_ALIGN_VARIABLE(4) = {0};
uint8_t tag[GCM_BLOCK_LEN] BSP_ALIGN_VARIABLE(4) = {0};
uint8_t cipher_calculated[GCM_BLOCK_LEN * 2] BSP_ALIGN_VARIABLE(4) = {0};
uint8_t plain_calculated[GCM_BLOCK_LEN * 2] BSP_ALIGN_VARIABLE(4) = {0};
uint32_t input_length = 0;
uint32_t output_length = 0;
uint32_t input_length_tmp = 0;
uint32_t output_length_tmp = 0;
/* Prepare wrapped key data area */
uint8_t wrapped_key_value[RSIP_BYTE_SIZE_WRAPPED_KEY(RSIP_KEY_TYPE_AES_256)] BSP_ALIGN_VARIABLE(4);
rsip_wrapped_key_t wrapped_key =
{
.type = RSIP_KEY_TYPE_AES_256, .p_value = &wrapped_key_value
};
/* Initialize the driver */
err = R_RSIP_Open(&g_rsip_ctrl, &g_rsip_cfg);
assert(FSP_SUCCESS == err);
/* Generate a aes 256-bit key */
err = R_RSIP_KeyGenerate(&g_rsip_ctrl, &wrapped_key);
assert(FSP_SUCCESS == err);
/* Generate a nonce */
err = R_RSIP_RandomNumberGenerate(&g_rsip_ctrl, nonce);
assert(FSP_SUCCESS == err);
/* Encrypt a plaintext */
err = R_RSIP_AES_AEAD_Init(&g_rsip_ctrl, RSIP_AES_AEAD_MODE_GCM_ENC, &wrapped_key, nonce, GCM_NONCE_LEN);
assert(FSP_SUCCESS == err);
/* Input AAD */
err = R_RSIP_AES_AEAD_AADUpdate(&g_rsip_ctrl, aad, GCM_AAD_LEN);
assert(FSP_SUCCESS == err);
/* Input plaintext (one-shot) */
err = R_RSIP_AES_AEAD_Update(&g_rsip_ctrl, plain, GCM_TEXT_LEN, cipher_calculated, &output_length_tmp);
output_length += output_length_tmp;
assert(FSP_SUCCESS == err);
/* Output remaining text and generate a tag */
err = R_RSIP_AES_AEAD_Finish(&g_rsip_ctrl, cipher_calculated + output_length, &output_length_tmp, tag);
assert(FSP_SUCCESS == err);
/* Decrypt a ciphertext using the same key */
err = R_RSIP_AES_AEAD_Init(&g_rsip_ctrl, RSIP_AES_AEAD_MODE_GCM_DEC, &wrapped_key, nonce, GCM_NONCE_LEN);
assert(FSP_SUCCESS == err);
/* Input AAD */
err = R_RSIP_AES_AEAD_AADUpdate(&g_rsip_ctrl, aad, GCM_AAD_LEN);
assert(FSP_SUCCESS == err);
input_length = 0;
output_length = 0;
/* Input ciphertext (multi-shot) */
input_length_tmp = 10;
err = R_RSIP_AES_AEAD_Update(&g_rsip_ctrl,
cipher_calculated + input_length,
input_length_tmp,
plain_calculated + output_length,
&output_length_tmp);
input_length += input_length_tmp;
output_length += output_length_tmp;
assert(FSP_SUCCESS == err);
input_length_tmp = GCM_TEXT_LEN - input_length;
err = R_RSIP_AES_AEAD_Update(&g_rsip_ctrl,
cipher_calculated + input_length,
input_length_tmp,
plain_calculated + output_length,
&output_length_tmp);
output_length += output_length_tmp;
assert(FSP_SUCCESS == err);
/*
* Output remaining text and verify the tag
* Attention: Do not use the output plaintext by R_RSIP_AES_Cipher_DecryptUpdate()
* before R_RSIP_AES_Cipher_DecryptFinal() returns FSP_SUCCESS.
* The integrity is checked in this function.
*/
err = R_RSIP_AES_AEAD_Verify(&g_rsip_ctrl, plain_calculated + output_length, &output_length_tmp, tag, GCM_TAG_LEN);
assert(FSP_SUCCESS == err);
/* Compare plain and plain_calculated */
assert(0 == memcmp(plain, plain_calculated, GCM_TEXT_LEN));
/* Close the driver */
err = R_RSIP_Close(&g_rsip_ctrl);
assert(FSP_SUCCESS == err);
}

/* Generate a key pair of secp256r1 for peer (Virtually generated for local validation) */
err = R_RSIP_KeyPairGenerate(&g_rsip_ctrl, &wrapped_peer_public_key, &wrapped_peer_private_key);
assert(FSP_SUCCESS == err);
/* Generate a key pair of secp256r1 for Local device (Assume product with RSIP driver) */
err = R_RSIP_KeyPairGenerate(&g_rsip_ctrl, &wrapped_local_public_key, &wrapped_local_private_key);
assert(FSP_SUCCESS == err);
/* CA process :
* In the following, generate certificate and signature of peer public key */
/* Export raw peer public key (Qx|Qy) and store it in cert */
err = R_RSIP_PublicKeyExport(&wrapped_peer_public_key, raw_public_key);
assert(FSP_SUCCESS == err);
memcpy(cert.public_key_param_qx, &raw_public_key[0], SECP256R1_KEY_SIZE);
memcpy(cert.public_key_param_qy, &raw_public_key[SECP256R1_KEY_SIZE], SECP256R1_KEY_SIZE);
/* Generate the hash of cert and store it in hash */
err = R_RSIP_SHA_Compute(&g_rsip_ctrl, RSIP_HASH_TYPE_SHA256, (uint8_t *) &cert, sizeof(cert), hash);
assert(FSP_SUCCESS == err);
/* Generate the signature of hash with CA private key and store it in signature */
err = R_RSIP_ECDSA_Sign(&g_rsip_ctrl, &wrapped_ca_private_key, hash, signature);
assert(FSP_SUCCESS == err);
/* Local device process :
* In the following, verify the CA certificate, import peer public key, and generate shared secret */
/* Verify the signature of cert with CA public key */
err = R_RSIP_PKI_ECDSA_CertVerify(&g_rsip_ctrl, &wrapped_ca_public_key, hash, signature);
assert(FSP_SUCCESS == err);
/* Import verified peer public key with wrapped format */
err = R_RSIP_PKI_CertKeyImport(&g_rsip_ctrl,
(uint8_t *) &cert,
sizeof(cert),
cert.public_key_param_qx,
SECP256R1_KEY_SIZE,
cert.public_key_param_qy,
SECP256R1_KEY_SIZE,
RSIP_HASH_TYPE_SHA256,
&wrapped_verified_peer_public_key);
assert(FSP_SUCCESS == err);
/* Generate shared secret from the Local private key and the verified peer public key */
err = R_RSIP_ECDH_KeyAgree(&g_rsip_ctrl,
&wrapped_local_private_key,
&wrapped_verified_peer_public_key,
&wrapped_secret);
assert(FSP_SUCCESS == err);
/* Close the driver */
err = R_RSIP_Close(&g_rsip_ctrl);
assert(FSP_SUCCESS == err);
}

KDF-SHA Example

This is an example of One-Step Key Derivation (NIST SP 800-56C) using KDF-SHA.

#define COUNTER_BYTE_LEN (4)
#define FIXED_INFO_BYTE_LEN (16)
#define KDF_SHA256_DKM_BYTE_SIZE (64)
#define CONCAT_DKM_BYTE_SIZE KDF_SHA256_DKM_BYTE_SIZE * 2
#define WRAPPED_IV_BYTE_SIZE (36)
#define GCM_BLOCK_LEN (16)
#define GCM_TEXT_LEN (31)
#define GCM_AAD_LEN (8)
#define GCM_TAG_LEN (16)
void r_rsip_example_kdf_sha ()
{
fsp_err_t err;
/* Prepare data area for key pair and shared secret */
uint8_t wrapped_public_key_value[RSIP_BYTE_SIZE_WRAPPED_KEY(RSIP_KEY_TYPE_ECC_SECP256R1_PUBLIC)]
BSP_ALIGN_VARIABLE(4);
rsip_wrapped_key_t wrapped_public_key =
{
.type = RSIP_KEY_TYPE_ECC_SECP256R1_PUBLIC, .p_value = &wrapped_public_key_value
};
uint8_t wrapped_private_key_value[RSIP_BYTE_SIZE_WRAPPED_KEY(RSIP_KEY_TYPE_ECC_SECP256R1_PRIVATE)]
BSP_ALIGN_VARIABLE(4);
rsip_wrapped_key_t wrapped_private_key =
{
.type = RSIP_KEY_TYPE_ECC_SECP256R1_PRIVATE, .p_value = &wrapped_private_key_value
};
rsip_wrapped_secret_t wrapped_secret;
/* Prepare data area for wrapped DKM (Derived Keying Material) */
uint8_t wrapped_dkm1[CONCAT_DKM_BYTE_SIZE];
uint8_t wrapped_dkm2[KDF_SHA256_DKM_BYTE_SIZE];
rsip_wrapped_dkm_t * p_wrapped_dkm1 = (rsip_wrapped_dkm_t *) wrapped_dkm1;
rsip_wrapped_dkm_t * p_wrapped_dkm2 = (rsip_wrapped_dkm_t *) wrapped_dkm2;
/* Prepare wrapped key and wrapped IV data area that will be derived from the DKM */
uint8_t wrapped_derived_key_value[RSIP_BYTE_SIZE_WRAPPED_KEY(RSIP_BYTE_SIZE_WRAPPED_KEY_AES_256)]
BSP_ALIGN_VARIABLE(4);
rsip_wrapped_key_t wrapped_derived_key =
{
.type = RSIP_KEY_TYPE_AES_256, .p_value = &wrapped_derived_key_value
};
uint8_t wrapped_iv[WRAPPED_IV_BYTE_SIZE];
/* Input data for KDF-SHA operation */
uint32_t counter = 0;
static const uint8_t fixed_info[FIXED_INFO_BYTE_LEN] BSP_ALIGN_VARIABLE(4) =
{
0x00, 0x01, 0x02, 0x03, 0x04, 0x05, 0x06, 0x07, 0x08, 0x09, 0x0a, 0x0b, 0x0c, 0x0d, 0x0e, 0x0f,
};
/* Prepare a data area for AES-GCM operation (For validation check) */
static const uint8_t plain[GCM_TEXT_LEN] BSP_ALIGN_VARIABLE(4) =
{
0x52, 0x65, 0x6e, 0x65, 0x73, 0x61, 0x73, 0x20, 0x45, 0x6c, 0x65, 0x63, 0x74, 0x72, 0x6f, 0x6e,
0x69, 0x63, 0x73, 0x20, 0x43, 0x6f, 0x72, 0x70, 0x6f, 0x72, 0x61, 0x74, 0x69, 0x6f, 0x6e,
};
static const uint8_t aad[GCM_AAD_LEN] BSP_ALIGN_VARIABLE(4) =
{
0x00, 0x01, 0x02, 0x03, 0x04, 0x05, 0x06, 0x07,
};
uint8_t tag[GCM_TAG_LEN] BSP_ALIGN_VARIABLE(4) = {0};
uint8_t cipher_calculated[GCM_BLOCK_LEN * 2] BSP_ALIGN_VARIABLE(4) = {0};
uint8_t plain_calculated[GCM_BLOCK_LEN * 2] BSP_ALIGN_VARIABLE(4) = {0};
uint32_t output_length = 0;
uint32_t output_length_tmp = 0;
/* Initialize the driver */
err = R_RSIP_Open(&g_rsip_ctrl, &g_rsip_cfg);
assert(FSP_SUCCESS == err);
/* In the following, generate shared secret from the locally generated key pair
* For more accurate process of generating the shared secret, refer to the PKI example code */
/* Generate a key pair of secp256r1 to generate the shared secret */
err = R_RSIP_KeyPairGenerate(&g_rsip_ctrl, &wrapped_public_key, &wrapped_private_key);
assert(FSP_SUCCESS == err);
/* Generate shared secret from the private key and public key */
err = R_RSIP_ECDH_KeyAgree(&g_rsip_ctrl, &wrapped_private_key, &wrapped_public_key, &wrapped_secret);
assert(FSP_SUCCESS == err);
/* One-Step Key Derivation (NIST SP 800-56C)
* Generate DKM (Derived Keying Material) */
/* In the following, calculate K(1) = KDF-SHA[counter|Z|FixedInfo], where Z is shared secret */
counter = 0x00000000;
counter++;
/* Initialize KDF-SHA operation */
err = R_RSIP_KDF_SHA_Init(&g_rsip_ctrl, RSIP_HASH_TYPE_SHA256);
assert(FSP_SUCCESS == err);
/* Input counter as a plaintext */
err = R_RSIP_KDF_SHA_Update(&g_rsip_ctrl, (uint8_t *) &counter, COUNTER_BYTE_LEN);
assert(FSP_SUCCESS == err);
/* Input shared secret as a ciphertext */
err = R_RSIP_KDF_SHA_ECDHSecretUpdate(&g_rsip_ctrl, &wrapped_secret);
assert(FSP_SUCCESS == err);
/* Input Fixed Info as a plaintext */
err = R_RSIP_KDF_SHA_Update(&g_rsip_ctrl, fixed_info, FIXED_INFO_BYTE_LEN);
assert(FSP_SUCCESS == err);
/* Finalize KDF-SHA operation and generate DKM1 (Derived Keying Material) */
err = R_RSIP_KDF_SHA_Finish(&g_rsip_ctrl, p_wrapped_dkm1);
assert(FSP_SUCCESS == err);
/* In the following, calculate K(2) = KDF-SHA[counter|Z|FixedInfo], where Z is shared secret */
counter++;
/* Initialize KDF-SHA operation */
err = R_RSIP_KDF_SHA_Init(&g_rsip_ctrl, RSIP_HASH_TYPE_SHA256);
assert(FSP_SUCCESS == err);
/* Input counter as a plaintext */
err = R_RSIP_KDF_SHA_Update(&g_rsip_ctrl, (uint8_t *) &counter, COUNTER_BYTE_LEN);
assert(FSP_SUCCESS == err);
/* Input shared secret as a ciphertext */
err = R_RSIP_KDF_SHA_ECDHSecretUpdate(&g_rsip_ctrl, &wrapped_secret);
assert(FSP_SUCCESS == err);
/* Input Fixed Info as a plaintext */
err = R_RSIP_KDF_SHA_Update(&g_rsip_ctrl, fixed_info, FIXED_INFO_BYTE_LEN);
assert(FSP_SUCCESS == err);
/* Finalize KDF-SHA operation and generate DKM2 (Derived Keying Material) */
err = R_RSIP_KDF_SHA_Finish(&g_rsip_ctrl, p_wrapped_dkm2);
assert(FSP_SUCCESS == err);
/* In the following, concatenate K(1) and K(2) to obtain DKM */
/* Concatenate K(2) to the end of K(1), and obtain DKM as p_wrapped_dkm1 */
err = R_RSIP_KDF_DKMConcatenate(p_wrapped_dkm1, p_wrapped_dkm2, CONCAT_DKM_BYTE_SIZE);
assert(FSP_SUCCESS == err);
/* In the following, derive wrapped key and wrapped IV for AES operation from the DKM
* In One-Step Key Derivation, it is used from the first position of the concatenated DKM
* Here, it is used from the 16th position as an example of how to use this API */
/* Derive the AES-256 wrapped key from the 16th byte position of the concatenated DKM1 */
err =
R_RSIP_KDF_DerivedKeyImport(&g_rsip_ctrl, p_wrapped_dkm1, 16, &wrapped_derived_key);
assert(FSP_SUCCESS == err);
/* Derive the wrapped IV for AES from the 16th byte position of the concatenated DKM1 */
err = R_RSIP_KDF_DerivedIVWrap(&g_rsip_ctrl,
p_wrapped_dkm1,
RSIP_INITIAL_VECTOR_TYPE_AES_16_BYTE,
16,
NULL,
wrapped_iv);
assert(FSP_SUCCESS == err);
/* In the following, verify the validity of wrapped_derived_key and wrapped_iv using AES-GCM operation */
/* Initialize AES-GCM operation using wrapped_derived_key and wrapped_iv */
err =
R_RSIP_AES_AEAD_Init(&g_rsip_ctrl, RSIP_AES_AEAD_MODE_GCM_ENC_WRAPPED_IV, &wrapped_derived_key, wrapped_iv, 0);
assert(FSP_SUCCESS == err);
/* Input AAD */
err = R_RSIP_AES_AEAD_AADUpdate(&g_rsip_ctrl, aad, GCM_AAD_LEN);
assert(FSP_SUCCESS == err);
/* Input plaintext and output ciphertext */
err = R_RSIP_AES_AEAD_Update(&g_rsip_ctrl, plain, GCM_TEXT_LEN, cipher_calculated, &output_length_tmp);
assert(FSP_SUCCESS == err);
output_length += output_length_tmp;
/* Output remaining ciphertext and generate a tag */
err = R_RSIP_AES_AEAD_Finish(&g_rsip_ctrl, cipher_calculated + output_length, &output_length_tmp, tag);
assert(FSP_SUCCESS == err);
output_length = 0;
/* Initialize AES-GCM operation using derived wrapped key and wrapped IV above */
err =
R_RSIP_AES_AEAD_Init(&g_rsip_ctrl, RSIP_AES_AEAD_MODE_GCM_DEC_WRAPPED_IV, &wrapped_derived_key, wrapped_iv, 0);
assert(FSP_SUCCESS == err);
/* Input AAD */
err = R_RSIP_AES_AEAD_AADUpdate(&g_rsip_ctrl, aad, GCM_AAD_LEN);
assert(FSP_SUCCESS == err);
/* Input ciphertext and output plaintext */
err = R_RSIP_AES_AEAD_Update(&g_rsip_ctrl, cipher_calculated, GCM_TEXT_LEN, plain_calculated, &output_length_tmp);
assert(FSP_SUCCESS == err);
output_length += output_length_tmp;
/* Output remaining plaintext and verify a tag */
err = R_RSIP_AES_AEAD_Verify(&g_rsip_ctrl, plain_calculated + output_length, &output_length_tmp, tag, GCM_TAG_LEN);
assert(FSP_SUCCESS == err);
/* Compare the plaintext data before encryption with the decrypted data */
assert(0 == memcmp(plain, plain_calculated, GCM_TEXT_LEN));
/* Close the driver */
err = R_RSIP_Close(&g_rsip_ctrl);
assert(FSP_SUCCESS == err);
}
